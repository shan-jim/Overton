staging.linked_records schema

join table · rebuilt nightly by link_records_rebuild
CREATE TABLE staging.linked_records ( id INT PRIMARY KEY, -- mirrors qmpQuoteID for readability m1_quote_id INT NOT NULL, -- FK → staging.m1_quotes.qmpQuoteID m1_quote_rev INT NOT NULL DEFAULT 0, pk_wo_number VARCHAR(10), -- FK → staging.pk_work_orders.WO_NUMBER · nullable if no WO yet pk_wo_status VARCHAR(16), join_method VARCHAR(24), -- "quote_ref_direct" | "part_rev_inferred" | "manual" join_confidence DECIMAL(3,2), -- 0.00–1.00 first_linked_at TIMESTAMP NOT NULL, last_verified_at TIMESTAMP NOT NULL, -- Phase 3 columns (not populated yet) -- -- pd_deal_id VARCHAR(32) NULL, -- pd_po_number VARCHAR(32) NULL, INDEX idx_m1 (m1_quote_id, m1_quote_rev), INDEX idx_pk (pk_wo_number), INDEX idx_fresh (last_verified_at) );
